int iskeyword(char a[]){
	char verb[2][10] = {"hates","likes"};
	char action[10] = "they";
	char keywords[2][10] = {"if","then"};
	int i=0,j;
	for( j=0;j<2;j++){
		if(strcmp(verb[j],a) == 0){
			i = 1;
			break;
		}
	}
	if(strcmp(action,a) == 0){
		i = 2;
	}
	for( j=0;j<2;j++){
		if(strcmp(keywords[j],a) == 0){
			i = 3;
			break;
		}
	}
	return i;
}
